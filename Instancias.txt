from clase_persona import Persona
from clase_estudiante import Estudiante

print("--- Pruebas con la clase Persona (Guía) ---")
# Creación de una instancia de Persona con datos genéricos
persona = Persona("Nombre Completo", 20)
print(f"Instancia inicial: {persona}")

# Modificación de atributos usando setters
persona.nombre = "Nuevo Nombre"
persona.edad = 25
print(f"Instancia modificada: {persona}\n")


print("--- Pruebas con la clase Estudiante (Guía) ---")
# Creación de una instancia de Estudiante con datos genéricos
estudiante = Estudiante("Nombre Completo", 20, "Curso de Ejemplo", "Universidad")
print(f"Instancia inicial: {estudiante}")

# Modificación de atributos propios y heredados
estudiante.nombre = "Nombre Modificado"
estudiante.edad = 21
estudiante.curso = "Nuevo Curso de Ejemplo"
estudiante.universidad = "Nueva Universidad"
print(f"Instancia modificada: {estudiante}")
