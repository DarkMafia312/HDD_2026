class Persona:

    # Constructor
    def __init__(self, nombre, edad):
        # Usamos los setters para asegurar que las validaciones se ejecuten al crear el objeto
        self.nombre = nombre
        self.edad = edad

    # Propiedad y Setter para Nombre
    @property
    def nombre(self):
        return self.__nombre

    @nombre.setter
    def nombre(self, valor):
        if len(valor.strip()) > 0:  # Evita nombres vacíos o con puros espacios
            self.__nombre = valor
        else:
            raise ValueError("Nombre no válido")

    # Propiedad y Setter para Edad
    @property
    def edad(self):
        return self.__edad
    
    @edad.setter
    def edad(self, valor):
        if valor >= 0:
            self.__edad = valor
        else:
            raise ValueError("La edad debe ser positiva")

    def __str__(self):
        return f"{self.__nombre} ({self.__edad})"
