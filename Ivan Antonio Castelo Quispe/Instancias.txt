from clase_persona import Persona
from clase_estudiante import Estudiante

print("--- Pruebas con la clase Persona ---")
# Creación inicial de la instancia Persona
persona = Persona("Ivan Antonio", 24)
print(f"Instancia inicial: {persona}")

# Modificación usando setters con su nombre completo
persona.nombre = "Ivan Antonio Castelo Quispe"
persona.edad = 25
print(f"Instancia modificada: {persona}\n")


print("--- Pruebas con la clase Estudiante ---")

estudiante = Estudiante("Ivan Antonio", 24, "Herramientas de Desarrollo", "UTP")

# Modificación de atributos propios y heredados
estudiante.nombre = "Ivan Antonio Castelo Quispe"
