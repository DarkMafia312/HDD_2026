from clase_persona import Persona
from clase_estudiante import Estudiante

print("--- Pruebas con la clase Persona ---")
# Creación inicial de la instancia Persona
persona = Persona("Fernando Antonio", 21)
print(f"Instancia inicial: {persona}")

# Modificación de atributos usando los setters
persona.nombre = "Fernando Antonio Gutiérrez"
persona.edad = 22
print(f"Instancia modificada: {persona}\n")


print("--- Pruebas con la clase Estudiante ---")
# Creación inicial de la instancia Estudiante
estudiante = Estudiante("Fernando Antonio", 20, "Lenguajes de Programación", "UTP")
print(f"Instancia inicial: {estudiante}")

# Modificación de atributos heredados y propios
estudiante.nombre = "Fernando Antonio"
estudiante.edad = 21  # Cambio de edad
estudiante.curso = "Estructuras de Datos"  # Cambio de curso
estudiante.universidad = "UTP"
print(f"Instancia modificada: {estudiante}")
